VibeLife Developer Manual

Complete technical reference for building, extending, and understanding the VibeLife platform. Fastify + TypeScript server, Godot 4.6 client, real-time WebSocket protocol.

Backend

TypeScript Fastify 5.2, Node.js ESM, 50+ services, 40+ route files, dual-mode persistence

Client

GDScript Godot 4.6, 30+ module scripts, procedural rendering, WebSocket + HTTP networking

Protocol

WebSocket JSON messages, sequence-tracked events, region-scoped broadcasting, command/event unions

Tick Loops

The server runs several interval-based loops:

TypeScript
// Enemy AI loop — 1 second interval
startEnemyTickLoop();    // Updates AI states, movement, attacks for all enemies

// NPC AI loop — 2 second interval
startNpcTickLoop();      // Updates NPC behavior, patrol, dialogue

// Combat regen — 1 second interval
setInterval(() => {
  for (const [, regionAvatars] of avatarsByRegion) {
    for (const avatar of regionAvatars.values()) {
      regenTick(avatar.accountId);  // 2% HP/s, 3% Mana/s
    }
  }
}, 1000);

// Session cleanup — 60 second interval
setInterval(() => {
  const now = Date.now();
  for (const [token, session] of sessions) {
    if (session.expiresAt < now) sessions.delete(token);
  }
}, 60000);

// Presence auto-away — 60 second interval
setInterval(() => checkAutoAway(), 60000);  // Mark inactive > 10min as away

Session Management

TypeScript
// Session TTL: 8 hours
const SESSION_TTL_MS = 1000 * 60 * 60 * 8;

// Session structure
type Session = {
  token: string;          // UUID v4
  accountId: string;      // UUID v4
  avatarId: string;       // UUID v4
  displayName: string;
  regionId: string;
  role: "resident" | "admin";
  expiresAt: number;      // Date.now() + TTL
};

// Stored in global Map
const sessions = new Map<string, Session>();

// Token validation refreshes TTL
export function resolveToken(token: string): Session | null {
  const session = sessions.get(token);
  if (!session) return null;
  if (session.expiresAt < Date.now()) {
    sessions.delete(token);
    return null;
  }
  session.expiresAt = Date.now() + SESSION_TTL_MS; // Refresh
  return session;
}

Token Resolution in Routes

Tokens can be passed in three ways. Routes extract them using a consistent pattern:

TypeScript
// In route handler:
const token = request.body?.token
  ?? request.query?.token
  ?? request.headers.authorization?.replace("Bearer ", "");

const session = resolveToken(token);
if (!session) return reply.status(401).send({ error: "Invalid or expired token" });

Password Hashing

TypeScript
import { scrypt, randomBytes, timingSafeEqual } from "crypto";

// Hash: scrypt with 16-byte salt, 64-byte key
async function hashPassword(password: string): Promise<string> {
  const salt = randomBytes(16).toString("hex");
  const key = await scryptAsync(password, salt, 64);
  return salt + ":" + key.toString("hex");
}

// Verify: timing-safe comparison
async function verifyPassword(password: string, stored: string): Promise<boolean> {
  const [salt, hash] = stored.split(":");
  const key = await scryptAsync(password, salt, 64);
  return timingSafeEqual(Buffer.from(hash, "hex"), key);
}

Permission System

Build Permission

TypeScript
type BuildPermission = {
  allowed: boolean;
  parcel: Parcel | null;
  reason?: string;
};

export function getBuildPermission(
  regionId: string, accountId: string, x: number, z: number
): BuildPermission {
  const parcel = findParcelAt(regionId, x, z);
  if (!parcel)
    return { allowed: false, parcel: null, reason: "No parcel here" };
  if (parcel.tier === "public")
    return { allowed: true, parcel };
  if (parcel.ownerAccountId === accountId)
    return { allowed: true, parcel };
  if (parcel.collaboratorAccountIds.includes(accountId))
    return { allowed: true, parcel };
  if (!parcel.ownerAccountId)
    return { allowed: false, parcel, reason: "Claim this parcel first" };
  return { allowed: false, parcel, reason: `Owned by ${ownerName}` };
}

Voxel Permission

TypeScript
export function getVoxelPermission(
  regionId: string, accountId: string, x: number, z: number
): { allowed: boolean; reason?: string } {
  const parcel = findParcelAt(regionId, x, z);
  if (!parcel) return { allowed: true };               // Wilderness: open
  if (parcel.tier === "public") return { allowed: true };
  if (parcel.ownerAccountId === accountId) return { allowed: true };
  if (parcel.collaboratorAccountIds.includes(accountId)) return { allowed: true };
  return { allowed: false, reason: "Not your parcel" };
}

Combat Service

File: src/world/combat-service.ts

TypeScript
// Stat formulas (per level)
maxHp     = 100 + level * 20
maxMana   = 50  + level * 10
strength  = 10  + level * 2
defense   = 5   + level * 2
xpToNext  = level * 100

// Damage calculation
computeDamage(attacker, defenderDef, style):
  melee:  damage = max(1, strength * 1.0 - defenderDef)
  magic:  damage = max(1, strength * 1.5 - defenderDef), costs 20 mana
  critical: 10% chance → damage * 2

// Death handling
handlePlayerDeath(accountId):
  → Deduct 5% currency as "death_penalty" transaction
  → Reset HP/Mana to max
  → Increment deaths counter
  → Respawn at first parcel center
  → Broadcast "combat:respawn"

// Regen tick (called every 1s)
regenTick(accountId):
  hp  += floor(maxHp * 0.02)    // 2% per second
  mana += floor(maxMana * 0.03)  // 3% per second

// XP & leveling
awardXp(accountId, amount):
  → Add XP, check threshold
  → If xp >= xpToNext: level++, recalculate stats, restore HP/Mana
  → Broadcast "combat:level_up"
  → Max level: 100

WebSocket Protocol

Connection Flow

TypeScript
// Endpoint: GET /ws/regions/:regionId?token=TOKEN&lastSequence=SEQ
app.get("/ws/regions/:regionId", { websocket: true }, (socket, req) => {
  const { regionId } = req.params;
  const token = req.query.token;
  const lastSequence = parseInt(req.query.lastSequence ?? "0");

  const session = resolveToken(token);
  if (!session) { socket.close(4001, "Unauthorized"); return; }

  // Register peer for broadcasting
  addPeer(regionId, session.accountId, socket);

  // Send full state snapshot
  socket.send(JSON.stringify(buildSnapshot(regionId, lastSequence)));

  // Message handler
  socket.on("message", (raw) => {
    const cmd = JSON.parse(raw.toString());
    if (!isRegionCommand(cmd)) return;
    handleCommand(session, regionId, cmd, socket);
  });

  socket.on("close", () => {
    removePeer(regionId, session.accountId);
    broadcastRegion(regionId, { type: "avatar:left", ... });
  });
});

Command Dispatch

The handleCommand() function in server.ts uses a switch on cmd.type:

TypeScript
switch (cmd.type) {
  case "move":
    moveAvatar(session, cmd.x, cmd.y, cmd.z);
    broadcastRegion(regionId, { type: "avatar:moved", ... });
    break;

  case "chat":
    if (cmd.message.length > 180) return;
    handleChat(session, cmd.message);
    broadcastRegion(regionId, { type: "chat", ... });
    break;

  case "whisper":
    handleWhisper(session, cmd.targetDisplayName, cmd.message);
    break;

  case "emote":
    broadcastRegion(regionId, { type: "avatar:emote", emoteName: cmd.emoteName, ... });
    checkEmoteCombo(regionId, session, cmd.emoteName);
    break;

  case "combat:attack":
    const result = attackEnemy(session, cmd.targetId, cmd.attackStyle);
    broadcastRegion(regionId, { type: "combat:damage", ... });
    if (result.killed) { /* loot, XP, achievements */ }
    break;

  case "voxel:place_block":
    const perm = getVoxelPermission(regionId, session.accountId, cmd.x, cmd.z);
    if (!perm.allowed) { socket.send(error); return; }
    setBlock(regionId, cmd.x, cmd.y, cmd.z, cmd.blockTypeId);
    broadcastRegion(regionId, { type: "voxel:block_placed", ... });
    onBlockPlaced(session.accountId);  // Achievement tracking
    break;

  // ... radio:tune, radio:skip, sit, stand, typing, group_chat, voxel:break_block
}

Event Broadcasting

TypeScript
// src/world/region.ts
const peers = new Map<string, Map<string, WebSocket>>();  // regionId → (accountId → ws)
let regionSequences = new Map<string, number>();

export function broadcastRegion(regionId: string, event: RegionEvent): void {
  const regionPeers = peers.get(regionId);
  if (!regionPeers) return;
  event.sequence = nextRegionSequence(regionId);
  const payload = JSON.stringify(event);
  for (const ws of regionPeers.values()) {
    if (ws.readyState === WebSocket.OPEN) ws.send(payload);
  }
}

export function nextRegionSequence(regionId: string): number {
  const seq = (regionSequences.get(regionId) ?? 0) + 1;
  regionSequences.set(regionId, seq);
  return seq;
}

Sequence Tracking

Every region event includes a monotonically increasing sequence number. Clients pass lastSequence when connecting to detect missed events. The snapshot includes the current sequence so the client knows where it stands.

Architecture Patterns Summary

Single-Process In-Memory

All state lives in Maps. Zero-latency reads, single server process. Scale-out requires sharding by region.

Barrel Re-exports

store.ts re-exports all services. Routes import from store.js or _shared-state.js only.

Region-Scoped Broadcast

All WS events are scoped to a region. broadcastRegion(regionId, event) sends to all peers in that region.

Dual Persistence
